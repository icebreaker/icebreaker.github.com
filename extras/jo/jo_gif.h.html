 62 // Based on NeuQuant algorithm
 63 static void jo_gif_quantize(unsigned char *rgba, int rgbaSize, int sample, unsigned char *map, int numColors) {
 64     // defs for freq and bias
 65     const int intbiasshift = 16; /* bias for fractions */
 66     const int intbias = (((int) 1) << intbiasshift);
 67     const int gammashift = 10; /* gamma = 1024 */
 68     const int betashift = 10;
 69     const int beta = (intbias >> betashift); /* beta = 1/1024 */
 70     const int betagamma = (intbias << (gammashift - betashift));
 71 
 72     // defs for decreasing radius factor
 73     const int radiusbiasshift = 6; /* at 32.0 biased by 6 bits */
 74     const int radiusbias = (((int) 1) << radiusbiasshift);
 75     const int radiusdec = 30; /* factor of 1/30 each cycle */
 76 
 77     // defs for decreasing alpha factor
 78     const int alphabiasshift = 10; /* alpha starts at 1.0 */
 79     const int initalpha = (((int) 1) << alphabiasshift);
 80 
 81     // radbias and alpharadbias used for radpower calculation
 82     const int radbiasshift = 8;
 83     const int radbias = (((int) 1) << radbiasshift);
 84     const int alpharadbshift = (alphabiasshift + radbiasshift);
 85     const int alpharadbias = (((int) 1) << alpharadbshift);
 86 
 87     sample = sample < 1 ? 1 : sample > 30 ? 30 : sample;
 88     int network[256][3];
 89     int bias[256] = {}, freq[256];
 90     for(int i = 0; i < numColors; ++i) {
 91         // Put nurons evenly through the luminance spectrum.
 92         network[i][0] = network[i][1] = network[i][2] = (i << 12) / numColors;
 93         freq[i] = intbias / numColors;
 94     }
 95     // Learn
 96     {
 97         const int primes[5] = {499, 491, 487, 503};
 98         int step = 4;
 99         for(int i = 0; i < 4; ++i) {
100             if(rgbaSize > primes[i] * 4 && (rgbaSize % primes[i])) { // TODO/Error? primes[i]*4?
101                 step = primes[i] * 4;
102             }
103         }
104         sample = step == 4 ? 1 : sample;
105 
106         int alphadec = 30 + ((sample - 1) / 3);
107         int samplepixels = rgbaSize / (4 * sample);
108         int delta = samplepixels / 100;
109         int alpha = initalpha;
110         delta = delta == 0 ? 1 : delta;
111 
112         int radius = (numColors >> 3) * radiusbias;
113         int rad = radius >> radiusbiasshift;
114         rad = rad <= 1 ? 0 : rad;
115         int radSq = rad*rad;
116         int radpower[32];
117         for (int i = 0; i < rad; i++) {
118             radpower[i] = alpha * (((radSq - i * i) * radbias) / radSq);
119         }
120 
121         // Randomly walk through the pixels and relax neurons to the "optimal" target.
122         for(int i = 0, pix = 0; i < samplepixels;) {
123             int r = rgba[pix + 0] << 4;
124             int g = rgba[pix + 1] << 4;
125             int b = rgba[pix + 2] << 4;
126             int j = -1;
127             {
128                 // finds closest neuron (min dist) and updates freq
129                 // finds best neuron (min dist-bias) and returns position
130                 // for frequently chosen neurons, freq[k] is high and bias[k] is negative
131                 // bias[k] = gamma*((1/numColors)-freq[k])
132 
133                 int bestd = 0x7FFFFFFF, bestbiasd = 0x7FFFFFFF, bestpos = -1;
134                 for (int k = 0; k < numColors; k++) {
135                     int *n = network[k];
136                     int dist = abs(n[0] - r) + abs(n[1] - g) + abs(n[2] - b);
137                     if (dist < bestd) {
138                         bestd = dist;
139                         bestpos = k;
140                     }
141                     int biasdist = dist - ((bias[k]) >> (intbiasshift - 4));
142                     if (biasdist < bestbiasd) {
143                         bestbiasd = biasdist;
144                         j = k;
145                     }
146                     int betafreq = freq[k] >> betashift;
147                     freq[k] -= betafreq;
148                     bias[k] += betafreq << gammashift;
149                 }
150                 freq[bestpos] += beta;
151                 bias[bestpos] -= betagamma;
152             }
153 
154             // Move neuron j towards biased (b,g,r) by factor alpha
155             network[j][0] -= (network[j][0] - r) * alpha / initalpha;
156             network[j][1] -= (network[j][1] - g) * alpha / initalpha;
157             network[j][2] -= (network[j][2] - b) * alpha / initalpha;
158             if (rad != 0) {
159                 // Move adjacent neurons by precomputed alpha*(1-((i-j)^2/[r]^2)) in radpower[|i-j|]
160                 int lo = j - rad;
161                 lo = lo < -1 ? -1 : lo;
162                 int hi = j + rad;
163                 hi = hi > numColors ? numColors : hi;
164                 for(int jj = j+1, m=1; jj < hi; ++jj) {
165                     int a = radpower[m++];
166                     network[jj][0] -= (network[jj][0] - r) * a / alpharadbias;
167                     network[jj][1] -= (network[jj][1] - g) * a / alpharadbias;
168                     network[jj][2] -= (network[jj][2] - b) * a / alpharadbias;
169                 }
170                 for(int k = j-1, m=1; k > lo; --k) {
171                     int a = radpower[m++];
172                     network[k][0] -= (network[k][0] - r) * a / alpharadbias;
173                     network[k][1] -= (network[k][1] - g) * a / alpharadbias;
174                     network[k][2] -= (network[k][2] - b) * a / alpharadbias;
175                 }
176             }
177 
178             pix += step;
179             pix = pix >= rgbaSize ? pix - rgbaSize : pix;
180 
181             // every 1% of the image, move less over the following iterations.
182             if(++i % delta == 0) {
183                 alpha -= alpha / alphadec;
184                 radius -= radius / radiusdec;
185                 rad = radius >> radiusbiasshift;
186                 rad = rad <= 1 ? 0 : rad;
187                 radSq = rad*rad;
188                 for (j = 0; j < rad; j++) {
189                     radpower[j] = alpha * ((radSq - j * j) * radbias / radSq);
190                 }
191             }
192         }
193     }
194     // Unbias network to give byte values 0..255
195     for (int i = 0; i < numColors; i++) {
196         map[i*3+0] = network[i][0] >>= 4;
197         map[i*3+1] = network[i][1] >>= 4;
198         map[i*3+2] = network[i][2] >>= 4;
199     }
200 }

202 typedef struct {
203     FILE *fp;
204     int numBits;
205     unsigned char buf[256];
206     unsigned char idx;
207     unsigned tmp;
208     int outBits;
209     int curBits;
210 } jo_gif_lzw_t;
211 
212 static void jo_gif_lzw_write(jo_gif_lzw_t *s, int code) {
213     s->outBits |= code << s->curBits;
214     s->curBits += s->numBits;
215     while(s->curBits >= 8) {
216         s->buf[s->idx++] = s->outBits & 255;
217         s->outBits >>= 8;
218         s->curBits -= 8;
219         if (s->idx >= 255) {
220             putc(s->idx, s->fp);
221             fwrite(s->buf, s->idx, 1, s->fp);
222             s->idx = 0;
223         }
224     }
225 }

227 static void jo_gif_lzw_encode(unsigned char *in, int len, FILE *fp) {
228     jo_gif_lzw_t state = { fp, 9, { 0, }, 0, 0, 0, 0 };
229     int maxcode = 511;
230 
231     // Note: 30k stack space for dictionary =|
232     const int hashSize = 5003;
233     short codetab[hashSize];
234     int hashTbl[hashSize];
235     memset(hashTbl, 0xFF, sizeof(hashTbl));
236 
237     jo_gif_lzw_write(&state, 0x100);
238 
239     int free_ent = 0x102;
240     int ent = *in++;
241 CONTINUE:
242     while (--len) {
243         int c = *in++;
244         int fcode = (c << 12) + ent;
245         int key = (c << 4) ^ ent; // xor hashing
246         while(hashTbl[key] >= 0) {
247             if(hashTbl[key] == fcode) {
248                 ent = codetab[key];
249                 goto CONTINUE;
250             }
251             ++key;
252             key = key >= hashSize ? key - hashSize : key;
253         }
254         jo_gif_lzw_write(&state, ent);
255         ent = c;
256         if(free_ent < 4096) {
257             if(free_ent > maxcode) {
258                 ++state.numBits;
259                 if(state.numBits == 12) {
260                     maxcode = 4096;
261                 } else {
262                     maxcode = (1<<state.numBits)-1;
263                 }
264             }
265             codetab[key] = free_ent++;
266             hashTbl[key] = fcode;
267         } else {
268             memset(hashTbl, 0xFF, sizeof(hashTbl));
269             free_ent = 0x102;
270             jo_gif_lzw_write(&state, 0x100);
271             state.numBits = 9;
272             maxcode = 511;
273         }
274     }
275     jo_gif_lzw_write(&state, ent);
276     jo_gif_lzw_write(&state, 0x101);
277     jo_gif_lzw_write(&state, 0);
278     if(state.idx) {
279         putc(state.idx, fp);
280         fwrite(state.buf, state.idx, 1, fp);
281     }
282 }

284 static int jo_gif_clamp(int a, int b, int c) { return a < b ? b : a > c ? c : a; }

286 jo_gif_t jo_gif_start(const char *filename, short width, short height, short repeat, int numColors) {
287     numColors = numColors > 255 ? 255 : numColors < 2 ? 2 : numColors;
288     jo_gif_t gif = {};
289     gif.width = width;
290     gif.height = height;
291     gif.repeat = repeat;
292     gif.numColors = numColors;
293     gif.palSize = log2(numColors);
294 
295     gif.fp = fopen(filename, "wb");
296     if(!gif.fp) {
297         return gif;
298     }
299 
300     fwrite("GIF89a", 6, 1, gif.fp);
301     // Logical Screen Descriptor
302     fwrite(&gif.width, 2, 1, gif.fp);
303     fwrite(&gif.height, 2, 1, gif.fp);
304     putc(0xF0 | gif.palSize, gif.fp);
305     fwrite("\x00\x00", 2, 1, gif.fp); // bg color index (unused), aspect ratio
306     return gif;
307 }

309 void jo_gif_frame(jo_gif_t *gif, unsigned char * rgba, short delayCsec, bool localPalette) {
310     if(!gif->fp) {
311         return;
312     }
313     short width = gif->width;
314     short height = gif->height;
315     int size = width * height;
316 
317     unsigned char localPalTbl[0x300];
318     unsigned char *palette = gif->frame == 0 || !localPalette ? gif->palette : localPalTbl;
319     if(gif->frame == 0 || localPalette) {
320         jo_gif_quantize(rgba, size*4, 1, palette, gif->numColors);
321     }
322 
323     unsigned char *indexedPixels = (unsigned char *)malloc(size);
324     {
325         unsigned char *ditheredPixels = (unsigned char*)malloc(size*4);
326         memcpy(ditheredPixels, rgba, size*4);
327         for(int k = 0; k < size*4; k+=4) {
328             int rgb[3] = { ditheredPixels[k+0], ditheredPixels[k+1], ditheredPixels[k+2] };
329             int bestd = 0x7FFFFFFF, best = -1;
330             // TODO: exhaustive search. do something better.
331             for(int i = 0; i < gif->numColors; ++i) {
332                 int bb = palette[i*3+0]-rgb[0];
333                 int gg = palette[i*3+1]-rgb[1];
334                 int rr = palette[i*3+2]-rgb[2];
335                 int d = bb*bb + gg*gg + rr*rr;
336                 if(d < bestd) {
337                     bestd = d;
338                     best = i;
339                 }
340             }
341             indexedPixels[k/4] = best;
342             int diff[3] = { ditheredPixels[k+0] - palette[indexedPixels[k/4]*3+0], ditheredPixels[k+1] - palette[indexedPixels[k/4]*3+1], ditheredPixels[k+2] - palette[indexedPixels[k/4]*3+2] };
343             // Floyd-Steinberg Error Diffusion
344             // TODO: Use something better -- http://caca.zoy.org/study/part3.html
345             if(k+4 < size*4) {
346                 ditheredPixels[k+4+0] = (unsigned char)jo_gif_clamp(ditheredPixels[k+4+0]+(diff[0]*7/16), 0, 255);
347                 ditheredPixels[k+4+1] = (unsigned char)jo_gif_clamp(ditheredPixels[k+4+1]+(diff[1]*7/16), 0, 255);
348                 ditheredPixels[k+4+2] = (unsigned char)jo_gif_clamp(ditheredPixels[k+4+2]+(diff[2]*7/16), 0, 255);
349             }
350             if(k+width*4+4 < size*4) {
351                 for(int i = 0; i < 3; ++i) {
352                     ditheredPixels[k-4+width*4+i] = (unsigned char)jo_gif_clamp(ditheredPixels[k-4+width*4+i]+(diff[i]*3/16), 0, 255);
353                     ditheredPixels[k+width*4+i] = (unsigned char)jo_gif_clamp(ditheredPixels[k+width*4+i]+(diff[i]*5/16), 0, 255);
354                     ditheredPixels[k+width*4+4+i] = (unsigned char)jo_gif_clamp(ditheredPixels[k+width*4+4+i]+(diff[i]*1/16), 0, 255);
355                 }
356             }
357         }
358         free(ditheredPixels);
359     }
360     if(gif->frame == 0) {
361         // Global Color Table
362         fwrite(palette, 3*(1<<(gif->palSize+1)), 1, gif->fp);
363         if(gif->repeat >= 0) {
364             // Netscape Extension
365             fwrite("\x21\xff\x0bNETSCAPE2.0\x03\x01", 16, 1, gif->fp);
366             fwrite(&gif->repeat, 2, 1, gif->fp); // loop count (extra iterations, 0=repeat forever)
367             putc(0, gif->fp); // block terminator
368         }
369     }
370     // Graphic Control Extension
371     fwrite("\x21\xf9\x04\x00", 4, 1, gif->fp);
372     fwrite(&delayCsec, 2, 1, gif->fp); // delayCsec x 1/100 sec
373     fwrite("\x00\x00", 2, 1, gif->fp); // transparent color index (first byte), currently unused
374     // Image Descriptor
375     fwrite("\x2c\x00\x00\x00\x00", 5, 1, gif->fp); // header, x,y
376     fwrite(&width, 2, 1, gif->fp);
377     fwrite(&height, 2, 1, gif->fp);
378     if (gif->frame == 0 || !localPalette) {
379         putc(0, gif->fp);
380     } else {
381         putc(0x80|gif->palSize, gif->fp );
382         fwrite(palette, 3*(1<<(gif->palSize+1)), 1, gif->fp);
383     }
384     putc(8, gif->fp); // block terminator
385     jo_gif_lzw_encode(indexedPixels, size, gif->fp);
386     putc(0, gif->fp); // block terminator
387     ++gif->frame;
388     free(indexedPixels);
389 }

391 void jo_gif_end(jo_gif_t *gif) {
392     if(!gif->fp) {
393         return;
394     }
395     putc(0x3b, gif->fp); // gif trailer
396     fclose(gif->fp);
